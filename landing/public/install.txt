#!/usr/bin/env bash
# Codetrail — install helpers + scaffolder. Self-contained.
#
# Usage:
#   curl -fsSL https://codetrail.dipgle.com/install.txt | bash
#
# Requirements: macOS / Linux / Windows (native or WSL2) + git + Node 20+
#                + Claude Code
#
# This script:
#   1. Clones github.com/dipgle/codetrail into ~/.codetrail (or git pulls if present)
#   2. Adds `np` and `adopt` shell helpers to ~/.zshrc or ~/.bashrc
#   3. Prints a one-line "next step"
#
# What you get after install:
#   - `np my-app`     scaffolds ~/Documents/projects/AI/my-app and opens Claude Code
#   - `adopt`         scaffolds CLAUDE.md + docs/ + memory/ + logs/devlog.sqlite in pwd
#   - mcp/            two MCP server implementations (open source, MIT):
#                     project-agent-node/ — default, cross-platform (Node 20+)
#                     project-agent-rs/   — optional, faster (Rust toolchain)
#                     `startup.sh` auto-builds whichever toolchain is present.
#   - hooks/          PostToolUse + SessionStart hooks (you copy to ~/.claude/hooks/
#                     yourself; the installer does NOT touch ~/.claude/ globally)

set -e

REPO="https://github.com/dipgle/codetrail.git"
HOME_DIR="${CODETRAIL_HOME:-$HOME/.codetrail}"

# 1. Clone or update repo at $HOME_DIR
if [ -d "$HOME_DIR/.git" ]; then
    echo "→ Updating $HOME_DIR"
    git -C "$HOME_DIR" pull --ff-only --quiet
else
    echo "→ Cloning $REPO into $HOME_DIR"
    git clone --quiet "$REPO" "$HOME_DIR"
fi

# 2. Ensure scaffolder is executable
chmod +x "$HOME_DIR/template/startup.sh" 2>/dev/null || true
chmod +x "$HOME_DIR/hooks/"*.sh 2>/dev/null || true

# 3. Pick an rc file (prefer zsh; fall back to bash; create zsh if neither)
RC="$HOME/.zshrc"
if [ ! -f "$RC" ] && [ -f "$HOME/.bashrc" ]; then
    RC="$HOME/.bashrc"
elif [ ! -f "$RC" ]; then
    touch "$RC"
fi

# 4. Append helpers if not already present (idempotent)
if grep -q "# Codetrail helpers" "$RC" 2>/dev/null; then
    echo "→ helpers already present in $RC — skipping append."
else
    cat >> "$RC" <<EOF

# Codetrail helpers — auto-installed
export CODETRAIL_HOME="\${CODETRAIL_HOME:-$HOME_DIR}"

np() {
  [ -z "\$1" ] && { echo "Usage: np <name|path>"; return 1; }
  local DIR
  if [[ "\$1" == */* ]]; then DIR="\$1"
  else DIR="\$HOME/Documents/projects/AI/\$1"
  fi
  "\$CODETRAIL_HOME/template/startup.sh" "\$DIR" || return 1
  cd "\$DIR" && command -v claude >/dev/null 2>&1 && claude || echo "(install Claude Code, then: cd \"\$DIR\" && claude)"
}

adopt() {
  local DIR="\${1:-\$(pwd)}"
  [ ! -d "\$DIR" ] && { echo "Not a dir: \$DIR"; return 1; }
  "\$CODETRAIL_HOME/template/startup.sh" "\$DIR" || return 1
  echo "Next: cd \"\$DIR\" && claude"
}
EOF
    echo "✓ Installed helpers into $RC"
fi

echo ""
echo "Next steps:"
echo "  1. source $RC   (or open a new terminal)"
echo "  2. np my-app    (creates ~/Documents/projects/AI/my-app/ and opens Claude)"
echo "     adopt        (scaffolds current dir for Codetrail)"
echo ""
echo "Source: https://github.com/dipgle/codetrail"
echo "Docs:   https://codetrail.dipgle.com/"
